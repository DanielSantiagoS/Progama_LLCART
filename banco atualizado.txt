CREATE DATABASE churrascaria_db;
USE churrascaria_db;

CREATE TABLE usuarios (
    id_usuario INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    login VARCHAR(50) NOT NULL UNIQUE,
    senha_hash VARCHAR(255) NOT NULL,
    permissao ENUM('GARCOM','CAIXA','GERENTE','ADMIN') NOT NULL,
    ativo TINYINT(1) DEFAULT 1
);

CREATE TABLE mesas (
    id_mesa INT AUTO_INCREMENT PRIMARY KEY,
    numero INT NOT NULL UNIQUE,
    status ENUM('LIVRE','OCUPADA','RESERVADA','FECHANDO') DEFAULT 'LIVRE'
);

CREATE TABLE comandas (
    id_comanda INT AUTO_INCREMENT PRIMARY KEY,
    id_mesa INT,
    cliente VARCHAR(100),
    data_abertura DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    data_fechamento DATETIME,
    status ENUM('ABERTA','FECHADA','CANCELADA') DEFAULT 'ABERTA',
    FOREIGN KEY (id_mesa) REFERENCES mesas(id_mesa)
);

CREATE TABLE produtos (
    id_produto INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(150) NOT NULL,
    categoria ENUM('CHURRASCO','COZINHA','BEBIDAS','SOBREMESAS','OUTROS') NOT NULL,
    preco DECIMAL(10,2) NOT NULL,
    controla_estoque TINYINT(1) DEFAULT 0
);

CREATE TABLE pedidos (
    id_pedido INT AUTO_INCREMENT PRIMARY KEY,
    id_comanda INT NOT NULL,
    id_usuario INT NOT NULL,
    hora DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    status ENUM('NOVO','EM_PREPARO','PRONTO','ENTREGUE','CANCELADO') DEFAULT 'NOVO',
    FOREIGN KEY (id_comanda) REFERENCES comandas(id_comanda),
    FOREIGN KEY (id_usuario) REFERENCES usuarios(id_usuario)
);

CREATE TABLE itens_pedido (
    id_item INT AUTO_INCREMENT PRIMARY KEY,
    id_pedido INT NOT NULL,
    id_produto INT NOT NULL,
    quantidade INT NOT NULL,
    obs VARCHAR(255),
    valor_unitario DECIMAL(10,2) NOT NULL,
    status ENUM('NORMAL','CANCELADO') DEFAULT 'NORMAL',
    FOREIGN KEY (id_pedido) REFERENCES pedidos(id_pedido),
    FOREIGN KEY (id_produto) REFERENCES produtos(id_produto)
);

CREATE TABLE reservas (
    id_reserva INT AUTO_INCREMENT PRIMARY KEY,
    nome_cliente VARCHAR(100) NOT NULL,
    telefone VARCHAR(20),
    data_hora DATETIME NOT NULL,
    id_mesa INT,
    FOREIGN KEY (id_mesa) REFERENCES mesas(id_mesa)
);

CREATE TABLE fornecedores (
    id_fornecedor INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(150) NOT NULL,
    telefone VARCHAR(20),
    categoria VARCHAR(50),
    ultimo_pedido DATE
);

CREATE TABLE compras (
    id_compra INT AUTO_INCREMENT PRIMARY KEY,
    id_fornecedor INT NOT NULL,
    data_compra DATE NOT NULL,
    total DECIMAL(10,2),
    FOREIGN KEY (id_fornecedor) REFERENCES fornecedores(id_fornecedor)
);

CREATE TABLE compras_itens (
    id_item INT AUTO_INCREMENT PRIMARY KEY,
    id_compra INT NOT NULL,
    id_produto INT NOT NULL,
    quantidade INT NOT NULL,
    valor_unitario DECIMAL(10,2) NOT NULL,
    FOREIGN KEY (id_compra) REFERENCES compras(id_compra),
    FOREIGN KEY (id_produto) REFERENCES produtos(id_produto)
);

CREATE TABLE estoque (
    id_estoque INT AUTO_INCREMENT PRIMARY KEY,
    id_produto INT NOT NULL,
    quantidade INT NOT NULL DEFAULT 0,
    FOREIGN KEY (id_produto) REFERENCES produtos(id_produto)
);

CREATE TABLE movimentacoes_estoque (
    id_mov INT AUTO_INCREMENT PRIMARY KEY,
    id_produto INT NOT NULL,
    quantidade INT NOT NULL,
    tipo ENUM('ENTRADA','SAIDA') NOT NULL,
    data_mov DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    origem ENUM('COMPRA','PEDIDO','AJUSTE') NOT NULL,
    FOREIGN KEY (id_produto) REFERENCES produtos(id_produto)
);

CREATE TABLE logs (
    id_log INT AUTO_INCREMENT PRIMARY KEY,
    id_usuario INT,
    acao VARCHAR(200),
    descricao TEXT,
    data_log DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (id_usuario) REFERENCES usuarios(id_usuario)
);
